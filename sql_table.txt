CREATE TABLE list(
list_id int unsigned auto_increment,
link_left int(10) DEFAULT '0' NOT NULL,
link_right int(10) DEFAULT '0' NOT NULL,
list varchar(50),
card_num int(10) DEFAULT '0' NOT NULL,
primary key (list_id)
);


CREATE TABLE card (
list_id int(10) DEFAULT '0' NOT NULL,
card_id	int(10) unsigned auto_increment,
link_left int(10) DEFAULT '0' NOT NULL,
link_right int(10) DEFAULT '0' NOT NULL,
card varchar(50),
comment_num	int(10) DEFAULT '0' NOT NULL,
primary key (card_id)
);



CREATE TABLE comment (
list_id int(10) DEFAULT '0' NOT NULL,
card_id int(10) DEFAULT '0' NOT NULL,
comment_id int(10) DEFAULT '0' NOT NULL,
mess varchar(100) NOT NULL,
date char(10),
user_name varchar(10) NOT NULL
);


CREATE TABLE member(
mem_id varchar(20) DEFAULT '' NOT NULL,
mem_email varchar(50),
mem_password varchar(20) DEFAULT '' NOT NULL,
primary key (mem_id)
);
